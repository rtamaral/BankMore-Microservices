# Dockerfile para o projeto principal BankMore API
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS base
WORKDIR /app
EXPOSE 80
EXPOSE 443

FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copiar arquivos de projeto
COPY ["BankMore.Api.csproj", "./"]
RUN dotnet restore "BankMore.Api.csproj"

# Copiar todo o c√≥digo fonte
COPY . .
WORKDIR "/src"
RUN dotnet build "BankMore.Api.csproj" -c Release -o /app/build

FROM build AS publish
RUN dotnet publish "BankMore.Api.csproj" -c Release -o /app/publish /p:UseAppHost=false

FROM base AS final
WORKDIR /app
COPY --from=publish /app/publish .
ENTRYPOINT ["dotnet", "BankMore.Api.dll"]
